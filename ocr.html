"""
OCR= Optical character recognition / Reconocimiento optico de caracteres

Por ejemplo:
-Foto de documento
-Documento escaneado
-Un recibo
-Señal con letras (foto bien enfocada)

¿Qué hace?
Reconoce la forma de las letras y convierte el texto digital editable/copiable

C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR
"""

import cv2              #Importa pythoncv para manipular
import pytesseract      #Libreria OCR

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

img_texto=cv2.imread("frases_motivadoras.webp")
img_redimensionar=cv2.resize(img_texto,(800,400))   #Redimensionar el tamaño de la imagen

#Convertir a escala de grises
img_gris=cv2.cvtColor(img_redimensionar,cv2.COLOR_BGR2GRAY)

#Seperamos el fondo con el texto
img_res=cv2.threshold(img_gris,120,225,cv2.THRESH_BINARY)[1]
#Devuelve dos valores; los pixeles y la imagen procesada

#Configurar OCR
texto=pytesseract.image_to_string(img_res,lang='spa')
#Como parametro le pasamos la variable con la imagen procesada y el idioma

print("Texto detectado")    #Imprimimos el titulo
print(texto)

cv2.imshow("Imagen analizada",img_redimensionar)
cv2.waitKey(0)
cv2.destroyAllWindows()

Glosario de Funciones

import pytesseract
Libreria ocr que sirve para reconocer caracteres
cv2.resize
Redimensiona la imagen
cv2.cvtColor
Cambia el color de la imagen
cv2.imread()
Función de OpenCV que lee una imagen desde un archivo. Soporta múltiples formatos como JPG, PNG, BMP, etc.
cv2.imshow()
Función de OpenCV que muestra una imagen en una ventana nueva. Requiere cv2.waitKey() para mantener la ventana abierta.
cv2.destroyAllWindows()
Cierra todas las ventanas de OpenCV que están abiertas. Útil para liberar recursos después de mostrar imágenes.
