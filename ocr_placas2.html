import cv2
import pytesseract

pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

# Lista de placas autorizadas
placas_autorizadas = ['XYZ789', 'CHS-09-03', 'VRZ-12-53', 'CDM-15-07', 'DLP876','FGR234','HJR675']

# Lista de imágenes (2 reales + 2 de simulación)
imagenes_placas = ["placas.jpg", "placa2.png", "placa3.PNG", "placa4.PNG"]

# Lista para guardar placas leídas
placas_leidas = []

# Procesar cada imagen
for img in imagenes_placas:
    # Cargar y redimensionar
    placa = cv2.imread(img)
    placanueva = cv2.resize(placa, (800, 300))

    # Convertir a escala de grises
    placa_gris = cv2.cvtColor(placanueva, cv2.COLOR_BGR2GRAY)

    # Separar fondo y texto
    placa_res = cv2.threshold(placa_gris, 120, 255, cv2.THRESH_BINARY)[1]

    # Leer texto con OCR
    texto_placa = pytesseract.image_to_string(placa_res, lang='spa')

    # Limpiar texto
    texto_placa = texto_placa.strip().replace(" ", "").replace("\n", "")

    # Guardar placa leída
    placas_leidas.append(texto_placa)

    # Validar si está autorizada
    if texto_placa in placas_autorizadas:
        print(f"ACCESO PERMITIDO: {texto_placa}")
    else:
        print(f"ACCESO DENEGADO: {texto_placa}")

# Mostrar placas leídas
print("\nPlacas leídas:")
for p in placas_leidas:
    print("-", p)

# Mostrar placas autorizadas que no se escanearon
print("\nPlacas autorizadas que no se escanearon:")
for p in placas_autorizadas:
    if p not in placas_leidas:
        print("-", p)

# Mostrar imágenes originales
for img in imagenes_placas:
    placa = cv2.imread(img)
    placanueva = cv2.resize(placa, (800, 300))
    cv2.imshow("Placa", placanueva)
    cv2.waitKey(0)

cv2.destroyAllWindows()

Glosario de Funciones

libreria pyteserect
Libreria ocr que sirve para reconocer caracteres
cv2.resize
Redimensiona la imagen
pytesseract.image_to_string
Devuelve la salida sin modificar como cadena de texto del procesamiento OCR de Tesseract.
cv2.imread()
Función de OpenCV que lee una imagen desde un archivo. Soporta múltiples formatos como JPG, PNG, BMP, etc.
cv2.imshow()
Función de OpenCV que muestra una imagen en una ventana nueva. Requiere cv2.waitKey() para mantener la ventana abierta.
cv2.destroyAllWindows()
Cierra todas las ventanas de OpenCV que están abiertas. Útil para liberar recursos después de mostrar imágenes.
