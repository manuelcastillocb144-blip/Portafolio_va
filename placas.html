import cv2
import pytesseract

# Ruta del ejecutable de Tesseract
pytesseract.pytesseract.tesseract_cmd = r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

# 1. Cargar imagen
img = cv2.imread("placa1.jpg")

# 2. Obtener tamaño
alto, ancho, _ = img.shape

# 3. Zona del recorte (letras grandes)
y1 = int(alto * 0.15)
y2 = int(alto * 0.75)
x1 = int(ancho * 0.05)
x2 = int(ancho * 0.95)
recorte = img[y1:y2, x1:x2]

# --- PREPROCESAMIENTO MEJORADO ---

# Convertir a gris
gris = cv2.cvtColor(recorte, cv2.COLOR_BGR2GRAY)

# Aumentar contraste (truco muy efectivo)
gris = cv2.convertScaleAbs(gris, alpha=2.0, beta=0)

# Eliminar ruido
blur = cv2.GaussianBlur(gris, (5, 5), 0)

# Umbral Otsu (mejor para texto grueso)
_, binaria = cv2.threshold(blur, 0, 255, cv2.THRESH_BINARY_INV + cv2.THRESH_OTSU)

# Aumentar tamaño antes de OCR
binaria_grande = cv2.resize(binaria, None, fx=2, fy=2, interpolation=cv2.INTER_CUBIC)

# --- OCR con whitelist y modo "texto sencillo" ---
config = r'-c tessedit_char_whitelist=ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789 --psm 7'

texto = pytesseract.image_to_string(binaria_grande, config=config)

# Limpiar texto
placa = texto.strip().replace(" ", "").replace("\n", "")

print("Placa detectada:", placa)

# Mostrar ventanas
cv2.imshow("Recorte", recorte)
cv2.imshow("Binaria", binaria)
cv2.imshow("Binaria Grande", binaria_grande)
cv2.waitKey(0)

Glosario de Funciones

cargar_imagen(ruta)
Carga una imagen desde la ruta del sistema de archivos especificada. Retorna un objeto de imagen de OpenCV o None si falla.
mostrar_imagen(imagen, titulo)
Despliega la imagen en una ventana con el título especificado. La ventana permanece abierta hasta que se presione una tecla.
obtener_dimensiones(imagen)
Extrae y retorna las dimensiones de la imagen como tupla: altura, ancho y número de canales de color.
cv2.imread()
Función de OpenCV que lee una imagen desde un archivo. Soporta múltiples formatos como JPG, PNG, BMP, etc.
cv2.imshow()
Función de OpenCV que muestra una imagen en una ventana nueva. Requiere cv2.waitKey() para mantener la ventana abierta.
cv2.destroyAllWindows()
Cierra todas las ventanas de OpenCV que están abiertas. Útil para liberar recursos después de mostrar imágenes.
